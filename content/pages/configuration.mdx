# Configuration

Muni is configured through the `muni.config.ts` file. This file contains all the settings for your documentation site.

## Basic Configuration

```typescript
const muniConfig: MuniConfig = {
  title: "Your Documentation",
  version: "1.0.0",
  defaultTheme: "minimal",
  showFrontmatterMeta: true,
  headingLinks: [
    { title: "GitHub", href: "https://github.com/your-org/your-repo" },
  ],
  navigation: [
    { title: "Introduction", slug: "index" },
    { title: "Getting Started", slug: "getting-started" },
  ],
  githubRepo: "https://github.com/your-org/your-repo",
};
```

## Configuration Options

### `title` (required)

The title of your documentation site.

```typescript
title: "My Documentation";
```

### `version` (optional)

The version of your documentation. Displayed in the header.

```typescript
version: "1.0.0";
```

### `defaultTheme` (optional)

The default theme for your site. Options: `"minimal"`, `"minimal-dark"`, `"mono"`.

```typescript
defaultTheme: "minimal";
```

### `showFrontmatterMeta` (optional)

Whether to show frontmatter metadata (author, date) at the bottom of pages.

```typescript
showFrontmatterMeta: true;
```

### `headingLinks` (optional)

Links displayed in the header next to the version.

```typescript
headingLinks: [
  { title: "GitHub", href: "https://github.com/your-org/your-repo" },
  { title: "Twitter", href: "https://twitter.com/your-handle" },
  { title: "Discord", href: "https://discord.gg/your-server" },
];
```

### `navigation` (required)

The navigation structure for your documentation.

```typescript
navigation: [
  { title: "Introduction", slug: "index" },
  { title: "Getting Started", slug: "getting-started" },
  {
    title: "API Reference",
    slug: "api/index",
    items: [
      { title: "Authentication", slug: "api/auth" },
      { title: "Users", slug: "api/users" },
    ],
  },
];
```

### `githubRepo` (optional)

The GitHub repository URL. Used for "Edit on GitHub" links.

```typescript
githubRepo: "https://github.com/your-org/your-repo";
```

## Navigation Structure

The navigation supports nested items for organizing your documentation:

```typescript
navigation: [
  // Simple page
  { title: "Home", slug: "index" },

  // Page with children
  {
    title: "Guides",
    slug: "guides/index",
    items: [
      { title: "Installation", slug: "guides/installation" },
      { title: "Configuration", slug: "guides/configuration" },
      { title: "Deployment", slug: "guides/deployment" },
    ],
  },

  // Deep nesting
  {
    title: "API",
    slug: "api/index",
    items: [
      {
        title: "Authentication",
        slug: "api/auth",
        items: [
          { title: "Login", slug: "api/auth/login" },
          { title: "Logout", slug: "api/auth/logout" },
        ],
      },
    ],
  },
];
```

## Environment Variables

You can use environment variables in your configuration:

```typescript
const muniConfig: MuniConfig = {
  title: process.env.NEXT_PUBLIC_SITE_TITLE || "Muni Docs",
  githubRepo: process.env.NEXT_PUBLIC_GITHUB_REPO,
  // ... other options
};
```

## Dynamic Configuration

For more complex setups, you can generate configuration dynamically:

```typescript
// Generate navigation from file system
const generateNavigation = () => {
  // Your logic here
  return [
    { title: "Home", slug: "index" },
    // ... more items
  ];
};

const muniConfig: MuniConfig = {
  title: "My Docs",
  navigation: generateNavigation(),
  // ... other options
};
```

## Validation

Muni validates your configuration at build time. Common errors:

- Missing required fields
- Invalid theme names
- Broken navigation links
- Invalid URL formats

Check the build output for validation errors.

## Best Practices

1. **Keep it simple**: Don't over-complicate your navigation structure
2. **Use consistent naming**: Keep slug names short and descriptive
3. **Group related content**: Use nested navigation for organization
4. **Test your links**: Make sure all navigation links work
5. **Version your docs**: Use the version field to track changes

## Advanced Configuration Patterns

### Multi-Environment Configuration

```typescript
const isDevelopment = process.env.NODE_ENV === 'development';
const isProduction = process.env.NODE_ENV === 'production';

const muniConfig: MuniConfig = {
  title: process.env.NEXT_PUBLIC_SITE_TITLE || "Muni Documentation",
  version: process.env.NEXT_PUBLIC_VERSION || "1.0.0",
  defaultTheme: (process.env.NEXT_PUBLIC_DEFAULT_THEME as Theme) || "minimal",
  showFrontmatterMeta: isDevelopment,
  headingLinks: [
    { title: "GitHub", href: process.env.NEXT_PUBLIC_GITHUB_URL || "#" },
    ...(isDevelopment ? [{ title: "Dev Tools", href: "/dev-tools" }] : []),
  ],
  navigation: generateNavigation(),
  githubRepo: process.env.NEXT_PUBLIC_GITHUB_REPO,
};
```

### Dynamic Navigation Generation

```typescript
import fs from 'fs';
import path from 'path';

function generateNavigation(): NavigationItem[] {
  const contentDir = path.join(process.cwd(), 'content/pages');
  const navigation: NavigationItem[] = [];
  
  function scanDirectory(dir: string, baseSlug = ''): void {
    const items = fs.readdirSync(dir);
    
    for (const item of items) {
      const fullPath = path.join(dir, item);
      const stat = fs.statSync(fullPath);
      
      if (stat.isDirectory()) {
        const slug = baseSlug ? `${baseSlug}/${item}` : item;
        const indexPath = path.join(fullPath, 'index.mdx');
        
        if (fs.existsSync(indexPath)) {
          const { data } = matter(fs.readFileSync(indexPath, 'utf-8'));
          navigation.push({
            title: data.title || item,
            slug,
            items: scanDirectory(fullPath, slug)
          });
        }
      } else if (item.endsWith('.mdx') && item !== 'index.mdx') {
        const slug = baseSlug ? `${baseSlug}/${item.replace('.mdx', '')}` : item.replace('.mdx', '');
        const { data } = matter(fs.readFileSync(fullPath, 'utf-8'));
        
        navigation.push({
          title: data.title || item.replace('.mdx', ''),
          slug
        });
      }
    }
  }
  
  scanDirectory(contentDir);
  return navigation;
}
```

### Type-Safe Configuration

```typescript
import { z } from 'zod';

const ConfigSchema = z.object({
  title: z.string().min(1),
  version: z.string().optional(),
  defaultTheme: z.enum(['minimal', 'minimal-dark', 'mono']).optional(),
  showFrontmatterMeta: z.boolean().optional(),
  headingLinks: z.array(z.object({
    title: z.string(),
    href: z.string().url()
  })).optional(),
  navigation: z.array(z.lazy(() => NavigationItemSchema)),
  githubRepo: z.string().url().optional(),
});

const NavigationItemSchema = z.object({
  title: z.string(),
  slug: z.string(),
  items: z.array(z.lazy(() => NavigationItemSchema)).optional(),
});

type MuniConfig = z.infer<typeof ConfigSchema>;

const muniConfig: MuniConfig = ConfigSchema.parse({
  title: "My Documentation",
  // ... rest of config
});
```
